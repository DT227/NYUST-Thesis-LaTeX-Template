% =====================================================================
% 1. 字型與基本中文設定 (CTeX)
% =====================================================================
\usepackage{fontspec}   % 設定字體 
\usepackage{xeCJK}      % 讓中英文字體分開設置 
\usepackage[UTF8,heading = true]{ctex}

%設定主要字型，也就是英文字型
\setmainfont[Mapping=tex-text]{Times New Roman}            
%設定中文字型
\setCJKmainfont[AutoFakeBold={4}]{DFKai-SB}
\XeTeXlinebreaklocale "zh"                %這兩行一定要加，中文才能自動換行 
\XeTeXlinebreakskip = 0pt plus 1pt        %這兩行一定要加，中文才能自動換行

% 設定目錄顯示層級與標題編號層級
\setcounter{tocdepth}{3}
\setcounter{secnumdepth}{3}


% =====================================================================
% 2. 基礎排版、表格與工具套件 (必須在 hyperref 之前)
% =====================================================================
\usepackage{geometry}   % 頁邊距設定
\usepackage{enumitem}   % 清單進階設定
\usepackage{multirow}   % 表格合併儲存格
\usepackage{diagbox}	  % 表格表頭畫斜線
\usepackage{graphicx}   % 插入圖片
\usepackage{lscape}     % 轉置表格
\usepackage{subcaption} % 進階圖片標題設定


% =====================================================================
% 3. 數學與演算法套件 (必須在 hyperref 之前！否則公式與演算法超連結會壞掉)
% =====================================================================
\usepackage{amsmath}    % 各式 AMS 數學功能
\usepackage{amssymb}    % 各式 AMS 數學符號
\usepackage{mathrsfs}   % 草寫體數學符號，在數學模式裡用 \mathscr{E} 得草寫 E
\usepackage{algorithmic}    % 演算法使用
\usepackage{algorithm}      % 演算法環境


% =====================================================================
% 4. 程式碼列表套件 (listings) 
% =====================================================================
\usepackage{listings}   % 程式列表套件
\lstset{
    breaklines=true,            % 過長的程式行可斷行
    extendedchars=false,        % 中文處理不需要 extendedchars
    texcl=true,                 % 中文註解需要有 TeX 處理過的 comment line, 所以設成 true
    comment=[l]\%\%,            % 以雙「百分號」做為程式中文註解的起頭標記，配合 MATLAB
    basicstyle=\ttfamily\small, % 小號字體, 約 10 pt 大小
    commentstyle=\upshape       % 預設是斜體字，會影響註解裏的英文，改用正體
    %language=Octave            % 會將一些 octave 指令以粗體顯示
}


% =====================================================================
% 5. 文獻引用與網址 (建議在 hyperref 之前)
% =====================================================================
\usepackage[nospace]{cite}  % 智慧文獻引用
\usepackage{url}            % 在文稿中引用網址，可以用 \url{http://www.ntust.edu.tw} 方式


% =====================================================================
% 6. 目錄外觀修改 (titletoc 區塊，保持原樣，放於 hyperref 之前)
% =====================================================================
\usepackage{titletoc}
\titlecontents{chapter}[0pt]
  {\addvspace{-0.3em}\bfseries}         %【目錄行距】\addvspace 控制與上一行的垂直距離
  {\thecontentslabel\ }                 % 保持原本的標號與間距（如：第一章 ）
  {}
  {\titlerule*[0.35pc]{.}\contentspage} %【點點間距】中括號內的數值越小，點點越緊密
  []
\titlecontents{section}[1.5em]          % 保持標準 Section 的左側縮排 (1.5em)
  {\addvspace{-0.3em}}                  %【目錄行距】與 Chapter 設為相同，即可達成等距
  {\thecontentslabel\ }                 % 保持原本的標號與間距（如：1.1 ）
  {}
  {\titlerule*[0.35pc]{.}\contentspage} %【點點間距】務必與 Chapter 保持一致
  []
\titlecontents{subsection}[3em]         % 保持標準 Subsection 的左側縮排 (3em)
  {\addvspace{-0.3em}}                  %【目錄行距】與上面保持一致
  {\thecontentslabel\ }                 % 保持原本的標號與間距（如：1.1.1 ）
  {}
  {\titlerule*[0.35pc]{.}\contentspage} %【點點間距】務必與上面保持一致
  []
\titlecontents{subsubsection}[4.5em]    % 保持標準 Subsubsection 的左側縮排 (4.5em)  
  {\addvspace{-0.3em}}                  %【目錄行距】與上面保持一致
  {\thecontentslabel\ }                 % 保持原本的標號與間距（如：1.1.1.1 ）
  {}
  {\titlerule*[0.35pc]{.}\contentspage} 
  []

\titlecontents{figure}[0pt]             % 編號的左側縮排空間對齊章節縮排
  {\addvspace{-0.3em}}                  % 保持與主目錄完全相同的固定等間距
  {圖~\thecontentslabel\ }              % 保持原本的圖編號格式
  {}
  {\titlerule*[0.35pc]{.}\contentspage} 
  []

\titlecontents{table}[0pt]              % 編號的左側縮排空間對齊章節縮排
  {\addvspace{-0.3em}}                  % 保持與主目錄完全相同的固定等間距
  {表~\thecontentslabel\ }              % 保持原本的表編號格式
  {}
  {\titlerule*[0.35pc]{.}\contentspage} 
  []


% =====================================================================
% 7. 圖表標題與雲科大專屬編號修改 (放於 hyperref 之前)
% =====================================================================
% 把所有 Figure caption 後的冒號給取消掉
% 參考https://www.ptt.cc/bbs/LaTeX/M.1186468410.A.476.html
\usepackage{caption}
\DeclareCaptionLabelSeparator{colon}{  }

% 將論文中的「表」、「圖」、「數學式」的編號格式，從預設的「小數點（.）」改為「連字號（-）」
\renewcommand\thetable{\arabic{chapter}-\arabic{table}}
\renewcommand\thefigure{\arabic{chapter}-\arabic{figure}}
% \renewcommand\theequation{\arabic{chapter}-\arabic{equation}}


% =====================================================================
% 8. 頁首頁尾與浮水印設定
% =====================================================================

% =====================================================================
% 9. 版面與全域規格設定
% =====================================================================
\geometry{verbose,a4paper,tmargin=3cm,bmargin=3cm,lmargin=3cm,rmargin=3cm}

\newcommand{\mybaselinestretch}{1.5}                %行距 1.5 倍 + 20%, (約為 double space)
\renewcommand{\baselinestretch}{\mybaselinestretch} % 論文行距預設值
\parskip=2ex        % 段落之間的間隔為兩個 x 的高度
\parindent=24Pt     % 段首內縮由 CJK 控制，所以這裡就設成不內縮

% 設定章節標題字體大小 (Chapter=14pt, Section=14pt, Subsection=14pt)
\ctexset{
    chapter = {
        format = \centering\bfseries\fontsize{14}{20}\selectfont,
        beforeskip = 1em,
        afterskip = 3em,
        fixskip = true,
    },
    section = {
        format = \bfseries\fontsize{14}{20}\selectfont, 
        beforeskip = 0.2em,
        afterskip = 0.2em,
    },
    subsection = {
        format = \bfseries\fontsize{14}{20}\selectfont, 
        beforeskip = 0.2em,
        afterskip = 0.2em,
    },
    subsubsection = {
        format = \bfseries\fontsize{14}{20}\selectfont, 
        beforeskip = 0.2em,
        afterskip = 0.2em,
    }
}

% 設定全域內文為 12pt
% 使用 \AtBeginDocument 確保這些設定在 \begin{document} 之後立即生效
% 這樣就不會被 Document Class 的預設值覆蓋
\AtBeginDocument{
    \fontsize{12}{18}\selectfont
}
